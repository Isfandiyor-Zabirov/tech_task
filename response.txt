                error: function (responce) {

        function subscribe(rewriteFilter) {

            var email = $('#subscrEmail').val();
            var name = $('#queryNameInSubscribe').val();
            var periodLen = $('#periodLen').val();
            var asFile = $('#cbAsFile').val();



            if (periodLen === '') {
                alert("Не указана длительность периода для подписки");
                return;
            }

            genXML(document.getElementById("filters"), document.getElementById("xmlData"), true, false);
            var xml = $('#xmlData').val();

            genXML(document.getElementById("filters"), document.getElementById("xmlData"), true, true);
            var xml4Subscription = $('#xmlData').val();

            genXML(document.getElementById("subscriptionParams"), document.getElementById("subscribeXmlData"), true, false);
            var subscribeXmlData = $('#subscribeXmlData').val();

            var data = {
                "email": email, "subscrName": name, "xml": xml, "asFile": asFile,
                "xml4Subscription": xml4Subscription, "scheduleparams": getSubscriptionsScheduleParams(),
                'rewriteFilter': rewriteFilter ? rewriteFilter : null,
                'subscribeXmlData': subscribeXmlData,
                'targetPageCode': targetPageCode
            };
             
            var SUBSCRIBE_URL = "/SearchQuery.aspx?name=Subscribe&PID=" + PID;

            $.post(SUBSCRIBE_URL, data).done(function (resp) {

                var responce = JSON.parse(resp);
                if (responce.result === "success") {

                    alert(responce.message);

                    closeModalWindow('#SubscribeWindow');
                }
                else if (responce.result === "error" && responce.status === "1") {
                    var confirmWindow = confirm("Для данного пользователя уже существует настройка с таким именем.\nПерезаписать настройку?");
                    if (confirmWindow == true) {
                        subscribe(true);
                    }
                }
                else {

                    alert(responce.message);
                }

            }).fail(function (responce) {

                alert('Возникла ошибка при отправке запроса на сервер');
                console.log(responce);

                closeModalWindow('#SubscribeWindow');

            });

        }

            function showTableOrGraph(obj)
            {
                var $this = $(obj);
                var img = $this.attr('src');
                if (img.indexOf('mybar.png') > 0)
                    img = img.replace('mybar.png', 'grid-three-up-24.jpg');
                else
                    img = img.replace('grid-three-up-24.jpg', 'mybar.png');
                $this.attr('src', img);
                var tabId = (obj.id.indexOf('timeLine') > 0 || obj.id.indexOf('procedureChart') > 0) ? obj.id.replace('img_', '') + 'Block' : obj.id.replace('img_', '') + 'Table';
                var graphId = (obj.id.indexOf('timeLine') > 0 || obj.id.indexOf('procedureChart') > 0) ? obj.id.replace('img_', '') + 'Area' : obj.id.replace('img_', '') + 'ChartArea';
                var $Tbl = $('#' + tabId);
                if ($Tbl.is(':visible'))
                {
                    $Tbl.hide();
                    $('#' + graphId).show();
                }
                else {
                    $Tbl.show();
                    $('#' + graphId).hide();
                }
            }


            function renderTableChart(tableId, arr) {
                var $tbody = $('#' + tableId).find('tbody');
                $tbody.empty();

                var prices = arr.map(function (s) {
                    return s.price_sums.value;
                });

                var maxDimensionTxt = '';
                if (prices.length > 0) {
                    var maxPrice = Math.max.apply(null, prices);
                    var maxDimension = sumFormatting(maxPrice);
                    maxDimensionTxt = maxDimension.substring(maxDimension.indexOf(' '), maxDimension.length);
                    maxDimension = Math.floor(Math.floor((Math.log(maxPrice) / Math.LN10)) / 3);
                    if (maxDimension > 4) maxDimension = 4;
                    else if (maxDimension < 1) maxDimensionTxt = '';
                }

                $tbody.next().find('td:contains("Цена")').text('Цена,' + maxDimensionTxt + ' руб.');
                arr.forEach(function (item) {
                    var p_s = item.price_sums.value / Math.pow(10, maxDimension * 3);
                    var tr = '<tr><td>' + item.key + '</td><td>' + item.doc_count + '</td><td>' + (Number(p_s).toFixed(2) > 0.01 ? Number(p_s).toFixed(2) : '< 0.01') + '</td></tr>';

                    $tbody.append(tr);
                });
            }

    function findOrganizatorInLongDict(obj) {
     
        var value = $('#orgLongDictInput').val();

        value = value.replace(/"/g, '\\"');

        var currPage = 0;

        if (obj) {
            currPage = parseInt($(obj).attr('content'));
            $('#orgCurrPage').val(currPage);
            currPage = currPage - 1;
        }

        var from = PAGE_SIZE * currPage;

        var data = { 'data': value, 'from': from };

        dataBus(data, LONG_DICTIONARY_URL, onSuccessOrganizatorLongDict, null);
    }

    function applyOrgFilter() {

        $('#' + GLOBAL_COMPOSITEFILTER_POINTER.idToPasteValue).find('.mustBeCleaned').empty();
        $('#' + GLOBAL_COMPOSITEFILTER_POINTER.idToPasteValue).find('.mustBeCleaned').val('');

        var $organizatorFilterWindow = $('#organizatorFilterWindow');

        $organizatorFilterWindow.css('display', 'none');
        var tabs = $organizatorFilterWindow.find('.tab-elem-org');
        var tabSize = tabs.size();

        var activeTab, tabindex;

        tabs.each(function (i, tab) {
            if ($(tab).hasClass('active')) {
                activeTab = $(tab);
                tabindex = i;
            }
        });

        var resultId = GLOBAL_COMPOSITEFILTER_POINTER.idToPasteValue;
        var $hiddenElemToPaste = $('#' + resultId).find('[escomposite-filter-hidden]');

        var orgSelectorWithContent;

        if (GLOBAL_COMPOSITEFILTER_POINTER.returnArray != null && GLOBAL_COMPOSITEFILTER_POINTER.returnArray.length >= tabSize) {
            var orgSelector = GLOBAL_COMPOSITEFILTER_POINTER.returnArray[tabindex];
            orgSelectorWithContent = '[content="node:'.concat(orgSelector, '"]');
        }

        var hiddenValueToCopy = activeTab.find('input[hidden-value]').val();
        setValueOrText($hiddenElemToPaste.find(orgSelectorWithContent).find('[content="leaf:value"]'), hiddenValueToCopy);

        var visibleContent = activeTab.find('[visible-value]').val();
        setOrgVisibleValue(visibleContent);

    }

    function setOrgVisibleValue(visibleContent) {
        var COLLAPSED_FILTER_NAME_LENGTH = 50;

        var resultId = GLOBAL_COMPOSITEFILTER_POINTER.idToPasteValue;
        var $visibleElem = $('#' + resultId).find('[esdict-visible-part]');

        if (visibleContent && visibleContent.length > 0) {
            var filterName = visibleContent.length > COLLAPSED_FILTER_NAME_LENGTH ? visibleContent.substr(0, COLLAPSED_FILTER_NAME_LENGTH) + "..." : visibleContent;
            setValueOrText($visibleElem, filterName);
        } else {
            setValueOrText($visibleElem, "");
        }

    }
    

    function onSuccessOrganizatorLongDict(responce) {

        if (JSON.parse(responce).result == "success") {

            var data = JSON.parse(JSON.parse(responce).data);
            var hits = data.hits.hits;

            var bodyContent = "";
            //TODO: вынести в метод на получение данных 
            var fields = ["buINN", "buKPP", "FullName"];

            hits.forEach(function (hint) {
                
                var buInnKppStr = hint._source["buInnKpp"] ? 'content="' + hint._source["buInnKpp"].toString() + '"' : "";

                bodyContent = bodyContent.concat('<tr data="' + hint._source["FullName"].toString().replace(new RegExp('"', 'g'), '&quot;') + '"' + buInnKppStr + ' onclick="addSelectedValueOfLongDictionary(this);">');

                var widths = [20, 20, 80];
                var i = 0;

                fields.forEach(function (code) {
                    bodyContent = bodyContent.concat('<td style="width:' + widths[i++] + '%">' + hint._source[code] + '</td>');
                });

                bodyContent = bodyContent.concat('</tr>');
            });

            $('#orgLongDictTable tbody').empty();
            $('#orgLongDictTable tbody').append(bodyContent);

            $('#longDictPagesControl').empty();

            var currPage = parseInt($('#orgCurrPage').val());
            if (!currPage) {
                currPage = 0;
            }

            var MAX_RESULTS_IN_RESPONCE = 10000;

            var totalAmount = data.hits.total.value;

            var totalPagerCount = Math.ceil(data.hits.total.value / PAGE_SIZE);

            if (totalAmount > MAX_RESULTS_IN_RESPONCE) {
                totalPagerCount = MAX_RESULTS_IN_RESPONCE / PAGE_SIZE;
            }

            totalPagerCount = totalPagerCount == 0 ? 1 : totalPagerCount;

            renderPager('longDictPagesControl', currPage, totalPagerCount, 'orgPageButton');

            $('#longDictWindow').show();

        }
    }

        document.OKDP2Dict.TypedWord = $('#findInDictInput').val();

                $row.find('td').eq(3).text(id);
